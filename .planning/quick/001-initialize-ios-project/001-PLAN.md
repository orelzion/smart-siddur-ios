       ```yaml
       name: SmartSiddur
       options:
         bundleIdPrefix: com.orelzion
         deploymentTarget:
           iOS: "17.0"
         xcodeVersion: "16.0"
         generateEmptyDirectories: true
       settings:
         base:
           SWIFT_VERSION: "6"
           MARKETING_VERSION: "1.0.0"
           CURRENT_PROJECT_VERSION: "1"

       packages:

         GoogleSignIn:
           url: https://github.com/google/GoogleSignIn-iOS
           from: "8.0.0"

       targets:
         SmartSiddur:
           type: application
           platform: iOS
           sources:
             - path: Sources
               excludes:
                 - "**/.gitkeep"
           settings:
             base:
               PRODUCT_BUNDLE_IDENTIFIER: com.orelzion.smartsiddur
               INFOPLIST_GENERATION_MODE: GeneratedFile
               INFOPLIST_KEY_UILaunchScreen_Generation: true
               INFOPLIST_KEY_UISupportedInterfaceOrientations: "UIInterfaceOrientationPortrait"
               INFOPLIST_KEY_CFBundleDisplayName: "SmartSiddur"
               SWIFT_STRICT_CONCURRENCY: complete
           dependencies:
             - package: Supabase
               product: Supabase
             - package: GoogleSignIn
               product: GoogleSignIn
       ```

       Key decisions in this spec:
       - Uses `from: "2.0.0"` for supabase-swift (v2.x is the current major, supports Swift 6 concurrency)
       - Uses `from: "8.0.0"` for GoogleSignIn (v8.x is current stable for Swift)
       - Swift strict concurrency set to `complete` to match Swift 6 expectations
       - Sources path is just `Sources` -- XcodeGen recursively includes all .swift files
       - .gitkeep files excluded from compilation
       - Generated Info.plist (no manual plist file needed)
